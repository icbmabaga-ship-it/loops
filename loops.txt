#include <iostream>
using namespace std;


int main() {
    int a[3][3];
    int rowSum, colSum;


    cout << "Enter 9 number:\n";
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            cin >> a[i][j];
        }
    }


    for (int i = 0; i < 3; i++) {
        rowSum = 0;
        for (int j = 0; j < 3; j++) {
            cout << a[i][j] << "\t";
            rowSum += a[i][j];
        }
        cout << "=  " << rowSum << endl;
    }
    cout << "----\t----\t----\n";
    for (int j = 0; j < 3; j++) {
        colSum = 0;
        for (int i = 0; i < 3; i++) {
            colSum += a[i][j];
        }
        cout << colSum << "\t";
    }


    return 0;
}
